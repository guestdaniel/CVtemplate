% Command to format two pieces of text, one left aligned and one right aligned
\newcommand{\aligngroup}[2]
{\begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
		#1 &
		#2 \\
\end{tabular*}}

% Command to format two pieces of text, one left aligned and one right aligned, with a bit of indentation on the left
% Horribly ugly way to do it but it works... should improve
\newcommand{\aligngroupin}[2]
{\setlength{\aligngroupinlen}{\linewidth}\addtolength{\aligngroupinlen}{-\step}\hspace*{\step}\begin{tabular*}{\aligngroupinlen}{@{}l@{\extracolsep{\fill}}r@{}}
		#1 &
		#2 \\
\end{tabular*}}

% indentsection environment, used to control indentation within sections 
\newenvironment{indentsection}%
	{\begin{list}%
			{}%
			{\setlength{\leftmargin}{\step}%
			 \setlength{\parsep}{\parskip}%
			 \setlength{\itemsep}{0pt}%
			}%
			\item[]%
	}%
	{\end{list}}%

% hangingpar environment, used to create hanging indent for bibliography
\newenvironment{hangingpar}%
	{\begin{list}%
			{}%
			{\setlength{\itemindent}{-\step}%
			 \setlength{\leftmargin}{\twostep}%
			 \setlength{\itemsep}{0pt}%
			 \setlength{\parsep}{\parskip}%
			 }%
			\setlength{\parindent}{-\step}%
			\item[]%
	}%
	{\end{list}
	 \vspace{0.25em}}%
